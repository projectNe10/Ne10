00051 /* factors buffer:
00052  * 0: stage number
00053  * 1: stride for the first stage
00054  * 2*stage number+2: algorithm flag to imply whether the generic algorithm works.
00055  * others: factors
00056  *
00057  * Only the leading 42 int32 is used to store factors.
00058  * The left can be used as algorithm flags, or status flags.
00059  * Even the leading bits of stage number can be reused.
00060  * */
00061 ne10_int32_t ne10_factor (ne10_int32_t n,
00062         ne10_int32_t * facbuf,
00063         ne10_int32_t ne10_factor_flags)
00064 {
00065     // This is a walk around. We need to "return" some flags.
00066     // Otherwise, we need to modify signature of ne10_factor.
00067     assert (NE10_MAXFACTORS >= 32);
00068 
00069     if ((facbuf == NULL)
00070         || (n <= 0))
00071     {
00072         return NE10_ERR;
00073     }
00074 
00075     ne10_int32_t p;
00076     ne10_int32_t i = 1;
00077     ne10_int32_t stage_num = 0;
00078     ne10_int32_t stride_max = n;
00079 
00080     // Default algorithm flag is NE10_FFT_ALG_24
00081     ne10_int32_t alg_flag = NE10_FFT_ALG_24;
00082 
00083     // Factor out powers of 4, 2, 5, 3, and other.
00084     do
00085     {
00086         // If NE10_FACTOR_FLAGS has enable NE10_FACTOR_EIGHT.
00087         // Try to combine one radix-4 and one radix-2 stages
00088         // into one radix-8 stage.
00089         if ((ne10_factor_flags & NE10_FACTOR_EIGHT)
00090                 && ((n==8) || (n==40) || (n==24)))
00091         {
00092             switch (n)
00093             {
00094             case 8:
00095                 p = 8;
00096                 break;
00097             case 40:
00098                 p = 5;
00099                 alg_flag = NE10_FFT_ALG_ANY;
00100                 break;
00101             case 24:
00102                 p = 3;
00103                 alg_flag = NE10_FFT_ALG_ANY;
00104                 break;
00105             }
00106         }
00107         else if ((n % 4) == 0)
00108         {
00109             p = 4;
00110         }
00111         else if ((n % 2) == 0)
00112         {
00113             p = 2;
00114         }
00115         else if ((n % 5) == 0)
00116         {
00117             p = 5;
00118             alg_flag = NE10_FFT_ALG_ANY;
00119         }
00120         else if ((n % 3) == 0)
00121         {
00122             p = 3;
00123             alg_flag = NE10_FFT_ALG_ANY;
00124         }
00125         else // stop factoring
00126         {
00127             p = n;
00128             alg_flag = NE10_FFT_ALG_ANY;
00129         }
00130 
00131         n /= p;
00132         facbuf[2 * i] = p;
00133         facbuf[2 * i + 1] = n;
00134         i++;
00135         stage_num++;
00136     }
00137     while (n > 1);
00138     facbuf[0] = stage_num;
00139     facbuf[1] = stride_max / p;
00140 
00141     if (stage_num > 21)
00142     {
00143         // Since nfft is ne10_int32_t, stage_num can never be greater than 21,
00144         // because 3^21 > 2^32
00145         return NE10_ERR;
00146     }
00147 
00148     facbuf[2 * i] = alg_flag;
00149     return NE10_OK;
00150 }

00152 // Twiddles matrix [mstride][radix-1]
00153 // First column (k == 0)is ignored because phase == 1, and
00154 // twiddle = (1.0, 0.0).
00155 static void ne10_fft_generate_twiddles_line_float32 (ne10_fft_cpx_float32_t * twiddles,
00156         const ne10_int32_t mstride,
00157         const ne10_int32_t fstride,
00158         const ne10_int32_t radix,
00159         const ne10_int32_t nfft)
00160 {
00161     ne10_int32_t j, k;
00162     ne10_float32_t phase;
00163     const ne10_float64_t pi = NE10_PI;
00164 
00165     for (j = 0; j < mstride; j++)
00166     {
00167         for (k = 1; k < radix; k++) // phase = 1 when k = 0
00168         {
00169             phase = -2 * pi * fstride * k * j / nfft;
00170             twiddles[mstride * (k - 1) + j].r = (ne10_float32_t) cos (phase);
00171             twiddles[mstride * (k - 1) + j].i = (ne10_float32_t) sin (phase);
00172         } // radix
00173     } // mstride
00174 }

00176 ne10_fft_cpx_float32_t* ne10_fft_generate_twiddles_float32 (ne10_fft_cpx_float32_t * twiddles,
00177         const ne10_int32_t * factors,
00178         const ne10_int32_t nfft )
00179 {
00180     ne10_int32_t stage_count = factors[0];
00181     ne10_int32_t fstride = factors[1];
00182     ne10_int32_t mstride;
00183     ne10_int32_t cur_radix; // current radix
00184 
00185     // for first stage
00186     cur_radix = factors[2 * stage_count];
00187     if (cur_radix % 2) // current radix is not 4 or 2
00188     {
00189         twiddles[0].r = 1.0;
00190         twiddles[0].i = 0.0;
00191         twiddles += 1;
00192         ne10_fft_generate_twiddles_line_float32 (twiddles, 1, fstride, cur_radix, nfft);
00193         twiddles += cur_radix - 1;
00194     }
00195     stage_count --;
00196 
00197     // for other stage
00198     for (; stage_count > 0; stage_count --)
00199     {
00200         cur_radix = factors[2 * stage_count];
00201         fstride /= cur_radix;
00202         mstride = factors[2 * stage_count + 1];
00203         ne10_fft_generate_twiddles_line_float32 (twiddles, mstride, fstride, cur_radix, nfft);
00204         twiddles += mstride * (cur_radix - 1);
00205     } // stage_count
00206 
00207     return twiddles;
00208 }

00257 
00258     // Can not factor.
00259     if (result == NE10_ERR)
00260     {
00261         NE10_FREE (st);
00262         return st;

00264 
00265     // Check if radix-8 can be enabled
00266     ne10_int32_t stage_count    = st->factors[0];
00267     ne10_int32_t algorithm_flag = st->factors[2 * (stage_count + 1)];
00268 
00269     // Enable radix-8.
00270     if (algorithm_flag == NE10_FFT_ALG_ANY)
00271     {
00272         result = ne10_factor (st->nfft, st->factors, NE10_FACTOR_EIGHT);
00273         if (result == NE10_ERR)

00279     }
00280     else
00281     {
00282         st->last_twiddles = NULL;
00283         st->nfft = nfft;
00284         result = ne10_factor (st->nfft, st->factors, NE10_FACTOR_DEFAULT);
00285         ne10_fft_generate_twiddles_float32 (st->twiddles, st->factors, st->nfft);
00286         return st;
00287     }
00288 
00289     // Generate super twiddles for the last stage.
00290     if (nfft % NE10_FFT_PARA_LEVEL == 0)
00291     {
00292         // Size of FFT satisfies requirement of NEON optimization.
00293         ne10_fft_generate_twiddles_line_float32 (st->last_twiddles,
00294                 st->nfft,

00296                 NE10_FFT_PARA_LEVEL,
