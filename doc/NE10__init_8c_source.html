00044 ne10_result_t ne10_init()
00045 {
00046     ne10_result_t status = NE10_ERR;
00047 #ifndef __MACH__
00048     FILE*   infofile = NULL;               // To open the file /proc/cpuinfo
00049     ne10_int8_t    cpuinfo[CPUINFO_BUFFER_SIZE];  // The buffer to read in the string
00050     ne10_uint32_t  bytes = 0;                     // Numbers of bytes read from the file
00051     ne10_int32_t     i = 0;                         // Temporary loop counter
00052 
00053     memset (cpuinfo, 0, CPUINFO_BUFFER_SIZE);
00054     infofile = fopen ("/proc/cpuinfo", "r");
00055 
00056     if (!infofile)
00057     {
00058         fprintf(stderr, "ERROR: couldn't read file \"/proc/cpuinfo\".\n");
00059         return NE10_ERR;
00060     }
00061 
00062     bytes    = fread (cpuinfo, 1, sizeof (cpuinfo), infofile);
00063     fclose (infofile);
00064 
00065     if (0 == bytes || CPUINFO_BUFFER_SIZE == bytes)
00066     {
00067         fprintf (stderr, "ERROR: Couldn't read the file \"/proc/cpuinfo\". NE10_init() failed.\n");
00068         return NE10_ERR;
00069     }
00070 
00071     while ('\0' != cpuinfo[i])
00072     {
00073         cpuinfo[i] = (ne10_int8_t) tolower (cpuinfo[i]);
00074         ++i;
00075     }
00076 
00077     if (0 != strstr ( (const char *)cpuinfo, "neon"))
00078     {
00079         is_NEON_available = NE10_OK;
00080     }
00081 #else  //__MACH__
00082     is_NEON_available = NE10_OK;
00083 #endif //__MACH__
00084 
00085 #if defined (NE10_ENABLE_MATH)
00086     status = ne10_init_math (is_NEON_available);
00087     if (status != NE10_OK)
00088     {
00089         fprintf(stderr, "ERROR: init math failed\n");
00090         return NE10_ERR;
00091     }
00092 #endif
00093 
00094 #if defined (NE10_ENABLE_DSP)
00095     status = ne10_init_dsp (is_NEON_available);
00096     if (status != NE10_OK)
00097     {
00098         fprintf(stderr, "ERROR: init dsp failed\n");

00100     }
00101 #endif
00102 
00103 #if defined (NE10_ENABLE_IMGPROC)
