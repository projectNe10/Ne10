00028 '''change the copy right year
00029 there are 4 cases
00030  1 2014 => 2014
00031  2 20xx => 20xx-14
00032  3 20xx-bb => 20xx-14
00033  4 20xx-14 => 20xx-14
00034 '''

00039 def extend_copyright(line, year):
00040     '''year is the year which you want to extend to'''
00041 
00042     #match the format like 'Copyright 2014 ARM Limited' or 'Copyright 2011-14 ARM Limited'
00043     p2014 = re.compile(r'.*{}.*'.format(year))
00044     if p2014.match(line):
00045         return line
00046 
00047     #match the format like 'Copyright 2011-12 ARM Limited'
00048     p20xx_bb = re.compile(r'(.*)(20\d\d)(-)(\d\d)(.*)')
00049     m = p20xx_bb.match(line)
00050     if m:
00051         return p20xx_bb.sub(r'\g<1>\g<2>\g<3>{}\g<5>'.format(year), line)
00052 
00053     #match the format like 'Copyright 2012 ARM Limited'
00054     p20xx = re.compile(r'(.*)(20\d\d)(.*)')
00055     m = p20xx.match(line)
00056     if m:
00057         return p20xx.sub(r'\g<1>\g<2>-{}\g<3>'.format(year), line)

00063         sys.stdout.write(line)

00086 def extend_copyright_all(extend_to_year):
00087     all_files = []
00088     for root, dirs, files in os.walk(os.getcwd()):
00089         for f in files:
00090             #exclude this script file
00091             if f != os.path.basename(sys.argv[0]):
00092                 all_files.append(os.path.join(root, f))
00093 
00094     pcopy_right = re.compile(r'.*Copyright [0-9-]* ARM Limited.*')
00095     for f in all_files:
00096         fd = open(f, 'r')
00097         for line in fd.readlines():
00098             m = pcopy_right.match(line)
00099             if m:
00100                 old_line = m.group(0)
00101                 new_line = extend_copyright(old_line, extend_to_year)
00102                 fd.close()
00103                 replace_line(f, old_line, new_line)
