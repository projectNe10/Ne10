00048 #define NE10_CPX_MUL_NEON_F32(Z,A,B) do {           \
00049     float32x4_t ARBR = vmulq_f32( A.val[0], B.val[0] ); \
00050     float32x4_t ARBI = vmulq_f32( A.val[0], B.val[1] ); \
00051     Z.val[0] = vmlsq_f32(ARBR, A.val[1], B.val[1]); \
00052     Z.val[1] = vmlaq_f32(ARBI, A.val[1], B.val[0]); \
00053 } while (0);

00055 #define NE10_CPX_MUL_INV_NEON_F32(Z,A,B) do {           \
00056     float32x4_t ARBR = vmulq_f32( A.val[0], B.val[0] ); \
00057     float32x4_t AIBI = vmulq_f32( A.val[1], B.val[1] ); \
00058     float32x4_t ARBI = vmulq_f32( A.val[0], B.val[1] ); \
00059     float32x4_t AIBR = vmulq_f32( A.val[1], B.val[0] ); \
00060     Z.val[0] = vaddq_f32(ARBR,AIBI);                \
00061     Z.val[1] = vsubq_f32(AIBR,ARBI);                \
00062 } while (0);
