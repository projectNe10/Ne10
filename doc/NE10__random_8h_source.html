00060 // a random number generator that generates IEEE 754 float numbers
00061 // NAN_OR_INF is to check whether the value is a NAN or an INF
00062 #define NAN_OR_INF (0xFF << 23)
00063 #define IS_NAN_OR_INF(x) ( ((x & NAN_OR_INF) == NAN_OR_INF)?1:0 )
00064 
00065 #define EXPONENT_MASK 0x807FFFFF
00066 #define IS_SUBNORMAL(x) ( ((x & EXPONENT_MASK) == x)?1:0 )
