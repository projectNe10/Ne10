47 static inline ne10_vec2f_t ne10_mul_matvec_float (ne10_mat2x2f_t T, ne10_vec2f_t v)
48 {
49  ne10_vec2f_t tmp;
50  ne10_float32_t x = (T.c2.r2 * v.x - T.c2.r1 * v.y) + T.c1.r1;
51  ne10_float32_t y = (T.c2.r1 * v.x + T.c2.r2 * v.y) + T.c1.r2;
52  tmp.x = x;
53  tmp.y = y;
54  return tmp;
55 }

94 void ne10_physics_compute_aabb_vec2f_c (ne10_mat2x2f_t *aabb,
95  ne10_vec2f_t *vertices,
96  ne10_mat2x2f_t *xf,
97  ne10_vec2f_t *radius,
98  ne10_uint32_t vertex_count)
99 {
100  ne10_vec2f_t lower = ne10_mul_matvec_float (*xf, vertices[0]);
101  ne10_vec2f_t upper = lower;
102  ne10_vec2f_t v;
103  ne10_int32_t i;
104 
105  for (i = 1; i < vertex_count; ++i)
106  {
107  v = ne10_mul_matvec_float (*xf, vertices[i]);
108  lower = min_2f (lower, v);
109  upper = max_2f (upper, v);
110  }
111 
112  aabb->c1.r1 = lower.x - radius->x;
113  aabb->c1.r2 = lower.y - radius->y;
114  aabb->c2.r1 = upper.x + radius->x;
115  aabb->c2.r2 = upper.y + radius->y;
116 
117 }

130 void ne10_physics_relative_v_vec2f_c (ne10_vec2f_t *dv,

139  ne10_vec2f_t vb;

141  for (i = 0; i < count; i++)
142  {
143  va.x = v_wa->x - v_wa->z * ra->y;
144  va.y = v_wa->y + v_wa->z * ra->x;
145  vb.x = v_wb->x - v_wb->z * rb->y;
146  vb.y = v_wb->y + v_wb->z * rb->x;

149  dv->y = vb.y - va.y;
