00044 ne10_float32_t cal_psnr_uint8_rgba (const ne10_uint8_t *pRef,
00045                                     const ne10_uint8_t *pTest,
00046                                     const ne10_uint32_t buffSize)
00047 {
00048     ne10_float64_t mse = 0.0, max = 255.0;
00049     ne10_uint32_t i;
00050     ne10_float32_t psnr_value;
00051 
00052     for (i = 0; i < buffSize; i++)
00053     {
00054         mse += (pRef[i] - pTest[i]) * (pRef[i] - pTest[i]);
00055     }
00056     mse = mse / buffSize / 4;
00057     psnr_value = 10 * log10 (max * max / mse);
00058     return psnr_value;
00059 }

00142 void boxfilter_get_kernels (size_t max_kernel_length,
00143                             ne10_size_t **kernels_ptr,
00144                             int *size)
00145 {
00146 
00147     if (max_kernel_length > BASIC_KERNEL_SIZE)
00148     {
00149         *size = KERNEL_COUNT + 3;
00150     }
00151     else if (max_kernel_length < BASIC_KERNEL_SIZE)
00152     {
00153         *size = max_kernel_length * max_kernel_length;
00154     }
00155     else
00156     {
00157         *size = KERNEL_COUNT;
00158     }
00159 
00160     *kernels_ptr = (ne10_size_t *) malloc (sizeof (ne10_size_t) * (*size));
00161     if (*kernels_ptr == 0)
00162     {
00163         printf ("**ERROR**: allocating %d bytes memory for kernels fails!\n",
00164                 sizeof (ne10_size_t) * (*size));
00165     }
00166 
00167     int x, y, first_part_size;
00168 
00169     if (max_kernel_length < BASIC_KERNEL_SIZE)
00170     {
00171         first_part_size = max_kernel_length;
00172     }
00173     else
00174     {
00175         first_part_size = BASIC_KERNEL_SIZE;
00176     }
00177 
00178     for (x = 0; x < first_part_size; x++)
00179     {
00180         for (y = 0; y < first_part_size; y++)
00181         {
00182             (*kernels_ptr) [x * first_part_size + y].x = x + 1;
00183             (*kernels_ptr) [x * first_part_size + y].y = y + 1;
00184         }
00185     }
00186     /* add:
00187      * max_kernel_length x 1
00188      * 1 x max_kernel_length
00189      * max_kernel_length x max_kernel_length
00190      * to kernels.
00191      */
00192     if (max_kernel_length > BASIC_KERNEL_SIZE)
00193     {
00194         (*kernels_ptr) [*size - 3].x = max_kernel_length;
00195         (*kernels_ptr) [*size - 3].y = 1;
00196         (*kernels_ptr) [*size - 2].x = 1;
00197         (*kernels_ptr) [*size - 2].y = max_kernel_length;
00198         (*kernels_ptr) [*size - 1].x = max_kernel_length;
00199         (*kernels_ptr) [*size - 1].y = max_kernel_length;
00200     }
00201 
00202     assert (valid_kernels (*kernels_ptr, *size) == NE10_OK);
00203 }
