00155      ** a second loop below computes the remaining 1 to 3 samples. */
00156     while (blkCnt > 0u)

00254             x3 = * (px++);

00327

00329 
00330     /* copy data */

00332     {

00334         *pStateCurnt++ = *pState++;

00344 
00345     /* Copy the remaining q31_t data */
00346     while (tapCnt > 0u)
00347     {
00348         *pStateCurnt++ = *pState++;
00349 
00350         /* Decrement the loop counter */
