116 void ne10_fir_float_c (const ne10_fir_instance_f32_t * S,
117  ne10_float32_t * pSrc,
118  ne10_float32_t * pDst,
119  ne10_uint32_t blockSize)
120 {
121 
122  ne10_float32_t *pState = S->pState; /* State pointer */
123  ne10_float32_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
124  ne10_float32_t *pStateCurnt; /* Points to the current sample of the state */
125  ne10_float32_t *px, *pb; /* Temporary pointers for state and coefficient buffers */
126  ne10_uint32_t numTaps = S->numTaps; /* Number of filter coefficients in the filter */
127  ne10_uint32_t i, tapCnt, blkCnt; /* Loop counters */
128 
129  /* Run the below code for Cortex-M4 and Cortex-M3 */
130 
131  ne10_float32_t acc0, acc1, acc2, acc3; /* Accumulators */
132  ne10_float32_t x0, x1, x2, x3, c0; /* Temporary variables to hold state and coefficient values */
133 
134 
135  /* S->pState points to state array which contains previous frame (numTaps - 1) samples */
136  /* pStateCurnt points to the location where the new input data should be written */
137  pStateCurnt = & (S->pState[ (numTaps - 1u)]);
138 
139  /* Apply loop unrolling and compute 4 output values simultaneously.
140  * The variables acc0 ... acc3 hold output values that are being computed:
141  *
142  * acc0 = b[numTaps-1] * x[n-numTaps-1] + b[numTaps-2] * x[n-numTaps-2] + b[numTaps-3] * x[n-numTaps-3] +...+ b[0] * x[0]
143  * acc1 = b[numTaps-1] * x[n-numTaps] + b[numTaps-2] * x[n-numTaps-1] + b[numTaps-3] * x[n-numTaps-2] +...+ b[0] * x[1]
144  * acc2 = b[numTaps-1] * x[n-numTaps+1] + b[numTaps-2] * x[n-numTaps] + b[numTaps-3] * x[n-numTaps-1] +...+ b[0] * x[2]
145  * acc3 = b[numTaps-1] * x[n-numTaps+2] + b[numTaps-2] * x[n-numTaps+1] + b[numTaps-3] * x[n-numTaps] +...+ b[0] * x[3]
146  */
147  blkCnt = blockSize >> 2;
148 
149  /* First part of the processing with loop unrolling. Compute 4 outputs at a time.
150  ** a second loop below computes the remaining 1 to 3 samples. */
151  while (blkCnt > 0u)
152  {
153  /* Copy four new input samples into the state buffer */
154  *pStateCurnt++ = *pSrc++;
155  *pStateCurnt++ = *pSrc++;
156  *pStateCurnt++ = *pSrc++;
157  *pStateCurnt++ = *pSrc++;
158 
159  /* Set all accumulators to zero */
160  acc0 = 0.0f;
161  acc1 = 0.0f;
162  acc2 = 0.0f;
163  acc3 = 0.0f;
164 
165  /* Initialize state pointer */
166  px = pState;
167 
168  /* Initialize coeff pointer */
169  pb = (pCoeffs);
170 
171  /* Read the first three samples from the state buffer: x[n-numTaps], x[n-numTaps-1], x[n-numTaps-2] */
172  x0 = *px++;
173  x1 = *px++;
174  x2 = *px++;
175 
176  /* Loop unrolling. Process 4 taps at a time. */
177  tapCnt = numTaps >> 2u;
178 
179  /* Loop over the number of taps. Unroll by a factor of 4.
180  ** Repeat until we've computed numTaps-4 coefficients. */
181  while (tapCnt > 0u)
182  {
183  /* Read the b[numTaps-1] coefficient */
184  c0 = * (pb++);
185 
186  /* Read x[n-numTaps-3] sample */
187  x3 = * (px++);
188 
189  /* acc0 += b[numTaps-1] * x[n-numTaps] */
190  acc0 += x0 * c0;
191 
192  /* acc1 += b[numTaps-1] * x[n-numTaps-1] */
193  acc1 += x1 * c0;
194 
195  /* acc2 += b[numTaps-1] * x[n-numTaps-2] */
196  acc2 += x2 * c0;
197 
198  /* acc3 += b[numTaps-1] * x[n-numTaps-3] */
199  acc3 += x3 * c0;
200 
201  /* Read the b[numTaps-2] coefficient */
202  c0 = * (pb++);
203 
204  /* Read x[n-numTaps-4] sample */
205  x0 = * (px++);
206 
207  /* Perform the multiply-accumulate */
208  acc0 += x1 * c0;
209  acc1 += x2 * c0;
210  acc2 += x3 * c0;
211  acc3 += x0 * c0;
212 
213  /* Read the b[numTaps-3] coefficient */
214  c0 = * (pb++);
215 
216  /* Read x[n-numTaps-5] sample */
217  x1 = * (px++);
218 
219  /* Perform the multiply-accumulates */
220  acc0 += x2 * c0;
221  acc1 += x3 * c0;
222  acc2 += x0 * c0;
223  acc3 += x1 * c0;
224 
225  /* Read the b[numTaps-4] coefficient */
226  c0 = * (pb++);
227 
228  /* Read x[n-numTaps-6] sample */
229  x2 = * (px++);
230 
231  /* Perform the multiply-accumulates */
232  acc0 += x3 * c0;
233  acc1 += x0 * c0;
234  acc2 += x1 * c0;
235  acc3 += x2 * c0;
236 
237  tapCnt--;
238  }
239 
240  /* If the filter length is not a multiple of 4, compute the remaining filter taps */
241  tapCnt = numTaps % 0x4u;
242 
243  while (tapCnt > 0u)
244  {
245  /* Read coefficients */
246  c0 = * (pb++);
247 
248  /* Fetch 1 state variable */
249  x3 = * (px++);
250 
251  /* Perform the multiply-accumulates */
252  acc0 += x0 * c0;
253  acc1 += x1 * c0;
254  acc2 += x2 * c0;
255  acc3 += x3 * c0;
256 
257  /* Reuse the present sample states for next sample */
258  x0 = x1;
259  x1 = x2;
260  x2 = x3;
261 
262  /* Decrement the loop counter */
263  tapCnt--;
264  }
265 
266  /* Advance the state pointer by 4 to process the next group of 4 samples */
267  pState = pState + 4;
268 
269  /* The results in the 4 accumulators, store in the destination buffer. */
270  *pDst++ = acc0;
271  *pDst++ = acc1;
272  *pDst++ = acc2;
273  *pDst++ = acc3;
274 
275  blkCnt--;
276  }
277 
278  /* If the blockSize is not a multiple of 4, compute any remaining output samples here.
279  ** No loop unrolling is used. */
280  blkCnt = blockSize % 0x4u;
281 
282  while (blkCnt > 0u)
283  {
284  /* Copy one sample at a time into state buffer */
285  *pStateCurnt++ = *pSrc++;
286 
287  /* Set the accumulator to zero */
288  acc0 = 0.0f;
289 
290  /* Initialize state pointer */
291  px = pState;
292 
293  /* Initialize Coefficient pointer */
294  pb = (pCoeffs);
295 
296  i = numTaps;
297 
298  /* Perform the multiply-accumulates */
299  do
300  {
301  acc0 += *px++ * *pb++;
302  i--;
303 
304  }
305  while (i > 0u);
306 
307  /* The result is store in the destination buffer. */
308  *pDst++ = acc0;
309 
310  /* Advance state pointer by 1 for the next sample */
311  pState = pState + 1;
312 
313  blkCnt--;
314  }
315 
316  /* Processing is complete.
317  ** Now copy the last numTaps - 1 samples to the satrt of the state buffer.
318  ** This prepares the state buffer for the next function call. */
319 
320  /* Points to the start of the state buffer */
321  pStateCurnt = S->pState;
322 
323  tapCnt = (numTaps - 1u) >> 2u;
324 
325  /* copy data */
326  while (tapCnt > 0u)
327  {
328  *pStateCurnt++ = *pState++;
329  *pStateCurnt++ = *pState++;
330  *pStateCurnt++ = *pState++;
331  *pStateCurnt++ = *pState++;
332 
333  /* Decrement the loop counter */
334  tapCnt--;
335  }
336 
337  /* Calculate remaining number of copies */
338  tapCnt = (numTaps - 1u) % 0x4u;
339 
340  /* Copy the remaining q31_t data */
341  while (tapCnt > 0u)
342  {
343  *pStateCurnt++ = *pState++;
344 
345  /* Decrement the loop counter */
346  tapCnt--;
347  }

949 void ne10_fir_lattice_float_c (const ne10_fir_lattice_instance_f32_t * S,
950  ne10_float32_t * pSrc,
951  ne10_float32_t * pDst,
952  ne10_uint32_t blockSize)
953 {
954  ne10_float32_t *pState; /* State pointer */
955  ne10_float32_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
956  ne10_float32_t *px; /* temporary state pointer */
957  ne10_float32_t *pk; /* temporary coefficient pointer */
958 
959 
960  /* Run the below code for Cortex-M4 and Cortex-M3 */
961 
962  ne10_float32_t fcurr1, fnext1, gcurr1, gnext1; /* temporary variables for first sample in loop unrolling */
963  ne10_float32_t fcurr2, fnext2, gnext2; /* temporary variables for second sample in loop unrolling */
964  ne10_float32_t fcurr3, fnext3, gnext3; /* temporary variables for third sample in loop unrolling */
965  ne10_float32_t fcurr4, fnext4, gnext4; /* temporary variables for fourth sample in loop unrolling */
966  ne10_uint32_t numStages = S->numStages; /* Number of stages in the filter */
967  ne10_uint32_t blkCnt, stageCnt; /* temporary variables for counts */
968 
969  gcurr1 = 0.0f;
970  pState = &S->pState[0];
971 
972  blkCnt = blockSize >> 2;
973 
974  /* First part of the processing with loop unrolling. Compute 4 outputs at a time.
975  a second loop below computes the remaining 1 to 3 samples. */
976  while (blkCnt > 0u)
977  {
978 
979  /* Read two samples from input buffer */
980  /* f0(n) = x(n) */
981  fcurr1 = *pSrc++;
982  fcurr2 = *pSrc++;
983 
984  /* Initialize coeff pointer */
985  pk = (pCoeffs);
986 
987  /* Initialize state pointer */
988  px = pState;
989 
990  /* Read g0(n-1) from state */
991  gcurr1 = *px;
992 
993  /* Process first sample for first tap */
994  /* f1(n) = f0(n) + K1 * g0(n-1) */
995  fnext1 = fcurr1 + ( (*pk) * gcurr1);
996  /* g1(n) = f0(n) * K1 + g0(n-1) */
997  gnext1 = (fcurr1 * (*pk)) + gcurr1;
998 
999  /* Process second sample for first tap */
1000  /* for sample 2 processing */
1001  fnext2 = fcurr2 + ( (*pk) * fcurr1);
1002  gnext2 = (fcurr2 * (*pk)) + fcurr1;
1003 
1004  /* Read next two samples from input buffer */
1005  /* f0(n+2) = x(n+2) */
1006  fcurr3 = *pSrc++;
1007  fcurr4 = *pSrc++;
1008 
1009  /* Copy only last input samples into the state buffer
1010  which will be used for next four samples processing */
1011  *px++ = fcurr4;
1012 
1013  /* Process third sample for first tap */
1014  fnext3 = fcurr3 + ( (*pk) * fcurr2);
1015  gnext3 = (fcurr3 * (*pk)) + fcurr2;
1016 
1017  /* Process fourth sample for first tap */
1018  fnext4 = fcurr4 + ( (*pk) * fcurr3);
1019  gnext4 = (fcurr4 * (*pk++)) + fcurr3;
