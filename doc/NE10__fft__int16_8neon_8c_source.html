61 {
62  if (fin == fout)
63  {
64  /* NOTE: for an in-place FFT algorithm. It just performs an out-of-place FFT into a temp buffer */
65  ne10_fft_cpx_int16_t * tmpbuf_ = (ne10_fft_cpx_int16_t*) NE10_MALLOC (sizeof (ne10_fft_cpx_int16_t) * nfft);
66 
67  // copy the data from input to output and bit reversal
68  ne10_data_bitreversal_int16 (tmpbuf_, fin, 1, &factors[2]);
69 
70  if (inverse_fft)
71  ne10_mixed_radix_butterfly_inverse_int16_neon (tmpbuf_, factors, twiddles);
72  else
73  ne10_mixed_radix_butterfly_int16_neon (tmpbuf_, factors, twiddles);
74 
75  memcpy (fout, tmpbuf_, sizeof (ne10_fft_cpx_int16_t) *nfft);
76  NE10_FREE (tmpbuf_);
77  }
78  else
79  {
80  // copy the data from input to output and bit reversal
81  ne10_data_bitreversal_int16 (fout, fin, 1, &factors[2]);
82 
83  if (inverse_fft)
84  ne10_mixed_radix_butterfly_inverse_int16_neon (fout, factors, twiddles);
85  else
86  ne10_mixed_radix_butterfly_int16_neon (fout, factors, twiddles);
87  }
88 }
