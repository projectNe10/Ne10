48  ne10_float32_t * pSrc,
49  ne10_uint32_t N,
50  ne10_float32_t * pReTable,
51  ne10_float32_t * pImTable,
52  ne10_float32_t * pDst)
53 {

55  ne10_float32_t *pCoefRe,*pCoefIm,*pOut1,*pIn1,*pOut2,*pIn2; /* Temporary pointers for twiddle factors */
56  ne10_uint32_t NBy2 = N>>1;

65  /*Mask value to select three entries*/
66  vmaskq_u32 = vld1q_u32(ne10_qMaskTable32+12);
67  vmask1q_u32 = vld1q_u32(ne10_qMaskTable32+4);

74  /*First Result*/

79  pDst[N+1] = 0;

87  vhalfq_f32 = vdupq_n_f32(0.5);
88 
89  /*for k=1 to N/4-1 and k=N/4+1 to K=N/2-1*/
90  for(k=0;k<Cnt;k++)
91  {
92  /*b[4] b[3] b[2] b[1] a[4] a[3] a[2] a[1]*/
93  vin1q2_f32 = vld2q_f32(pIn1);
94  pIn1+=8;
95  /*b[N/2-1] b[N/2-2] b[N/2-3] b[N/2-4] a[N/2-1] a[N/2-2] a[N/2-3] a[N/2-4]*/
96  vin2q2_f32 = vld2q_f32(pIn2);
97  pIn2-=8;
98  /* a[N/2-2] a[N/2-1] a[N/2-4] a[N/2-3]*/
99  vtmp1q_f32 =vrev64q_f32(vin2q2_f32.val[0]);
100  /* b[N/2-2] b[N/2-1] b[N/2-4] b[N/2-3]*/
101  vtmp2q_f32 =vrev64q_f32(vin2q2_f32.val[1]);
102  /*a[N/2-4] a[N/2-3] a[N/2-2] a[N/2-1]*/
103  vtmp1q_f32 = vcombine_f32(vget_high_f32(vtmp1q_f32),vget_low_f32(vtmp1q_f32));
104  /* b[N/2-4] b[N/2-3] b[N/2-2] b[N/2-1]*/
105  vtmp2q_f32 = vcombine_f32(vget_high_f32(vtmp2q_f32),vget_low_f32(vtmp2q_f32));
106

108  vureq_f32 = vaddq_f32(vin1q2_f32.val[0],vtmp1q_f32);

110  vuimq_f32 = vsubq_f32(vin1q2_f32.val[1],vtmp2q_f32);
111  /*VRe = (b[k]+b[N/2-k])/2*/
112  vvreq_f32 = vaddq_f32(vin1q2_f32.val[1],vtmp2q_f32);

117  vuimq_f32 = vmulq_f32(vuimq_f32,vhalfq_f32);
118  vvreq_f32 = vmulq_f32(vvreq_f32,vhalfq_f32);
119  vvimq_f32 = vmulq_f32(vvimq_f32,vhalfq_f32);

131  vtmp2q_f32 = vmlsq_f32(vtmp2q_f32,vvreq_f32,vimtwdq_f32);
132  //pDst[2*k] = uRe + reTmp;
133  //pDst[2*k+1] = uIm + imTmp;
134  vin1q2_f32.val[0] = vaddq_f32(vureq_f32,vtmp1q_f32);
135  vin1q2_f32.val[1] = vaddq_f32(vuimq_f32,vtmp2q_f32);
136  //pDst[2*(NBy2+k)] = uRe-reTmp;
137  //pDst[2*(NBy2+k)+1] = uIm-imTmp;
138  vin2q2_f32.val[0] = vsubq_f32(vureq_f32,vtmp1q_f32);
139  vin2q2_f32.val[1] = vsubq_f32(vuimq_f32,vtmp2q_f32);
140  vst2q_f32(pOut1,vin1q2_f32);
141  vst2q_f32(pOut1+N,vin2q2_f32);
142  pOut1+=8;
143 
144  //pDst[2*(NBy2-k)] = uRe-reTmp;
145  //pDst[2*(NBy2-k)+1] = imTmp-uIm;
146  vtmp2q_f32 = vsubq_f32(vtmp2q_f32,vuimq_f32);
147  vtmp1q_f32 = vrev64q_f32(vin2q2_f32.val[0]);
148  vtmp2q_f32 = vrev64q_f32(vtmp2q_f32);
149 
150  vin2q2_f32.val[0] = vcombine_f32(vget_high_f32(vtmp1q_f32),vget_low_f32(vtmp1q_f32));
151  vin2q2_f32.val[1] = vcombine_f32(vget_high_f32(vtmp2q_f32),vget_low_f32(vtmp2q_f32));
152 
153  //pDst[2*(N-k)] = uRe + reTmp;
154  //pDst[2*(N-k)+1] = -(uIm + imTmp);
155  vtmp2q_f32 = vnegq_f32(vin1q2_f32.val[1]);
156  vtmp1q_f32 = vrev64q_f32(vin1q2_f32.val[0]);
157  vtmp2q_f32 = vrev64q_f32(vtmp2q_f32);
158 
159  vin1q2_f32.val[0] = vcombine_f32(vget_high_f32(vtmp1q_f32),vget_low_f32(vtmp1q_f32));
160  vin1q2_f32.val[1] = vcombine_f32(vget_high_f32(vtmp2q_f32),vget_low_f32(vtmp2q_f32));
161 
162  vst2q_f32(pOut2,vin2q2_f32);
163  vst2q_f32(pOut2+N,vin1q2_f32);
164  pOut2-=8;
165  }
166  /*Lst Three VAlues*/
167  /*b[4] b[3] b[2] b[1] a[4] a[3] a[2] a[1]*/
168  vin1q2_f32 = vld2q_f32(pIn1);
169  pIn1+=8;
170  /*b[N/2-1] b[N/2-2] b[N/2-3] b[N/2-4] a[N/2-1] a[N/2-2] a[N/2-3] a[N/2-4]*/
171  vin2q2_f32 = vld2q_f32(pIn2);
172  pIn2-=8;
173  /* a[N/2-2] a[N/2-1] a[N/2-4] a[N/2-3]*/
174  vtmp1q_f32 =vrev64q_f32(vin2q2_f32.val[0]);
175  /* b[N/2-2] b[N/2-1] b[N/2-4] b[N/2-3]*/
176  vtmp2q_f32 =vrev64q_f32(vin2q2_f32.val[1]);
177  /*a[N/2-4] a[N/2-3] a[N/2-2] a[N/2-1]*/
178  vtmp1q_f32 = vcombine_f32(vget_high_f32(vtmp1q_f32),vget_low_f32(vtmp1q_f32));
179  /* b[N/2-4] b[N/2-3] b[N/2-2] b[N/2-1]*/
180  vtmp2q_f32 = vcombine_f32(vget_high_f32(vtmp2q_f32),vget_low_f32(vtmp2q_f32));
181 
182  /*uRe = (a[k]+a[N/2-k])/2*/
183  vureq_f32 = vaddq_f32(vin1q2_f32.val[0],vtmp1q_f32);
184  /*uIm = (b[k]-b[N/2-k])/2*/

186  /*VRe = (b[k]+b[N/2-k])/2*/
187  vvreq_f32 = vaddq_f32(vin1q2_f32.val[1],vtmp2q_f32);

189  vvimq_f32 = vsubq_f32(vtmp1q_f32,vin1q2_f32.val[0]);

197  vimtwdq_f32 = vld1q_f32(pCoefIm);

199  pCoefIm+=4;

202  vtmp1q_f32 = vmulq_f32(vvreq_f32,vretwdq_f32);
203  vtmp1q_f32 = vmlaq_f32(vtmp1q_f32,vvimq_f32,vimtwdq_f32);
204  /*imTmp = vIm*reTwd - vRe*imTwd */
205  vtmp2q_f32 = vmulq_f32(vvimq_f32,vretwdq_f32);

207  //pDst[2*k] = uRe + reTmp;
208  //pDst[2*k+1] = uIm + imTmp;

211 
212  vtmpq2_f32 = vld2q_f32(pOut1);
213  vin1q2_f32.val[0] = vbslq_f32(vmaskq_u32,vin1q2_f32.val[0],vtmpq2_f32.val[0]);
214  vin1q2_f32.val[1] = vbslq_f32(vmaskq_u32,vin1q2_f32.val[1],vtmpq2_f32.val[1]);
215  //pDst[2*(NBy2+k)] = uRe-reTmp;
216  //pDst[2*(NBy2+k)+1] = uIm-imTmp;
217  vin2q2_f32.val[0] = vsubq_f32(vureq_f32,vtmp1q_f32);
218  vin2q2_f32.val[1] = vsubq_f32(vuimq_f32,vtmp2q_f32);

225  vst2q_f32(pOut1+N,vin2q2_f32);

229  //pDst[2*(NBy2-k)] = uRe-reTmp;
230  //pDst[2*(NBy2-k)+1] = imTmp-uIm;
231  vtmp2q_f32 = vnegq_f32(vin2q2_f32.val[1]);

237 
238  vtmpq2_f32 = vld2q_f32(pOut2);
239  vin2q2_f32.val[0] = vbslq_f32(vmask1q_u32,vtmpq2_f32.val[0],vin2q2_f32.val[0]);

241 
242 
243  //pDst[2*(N-k)] = uRe + reTmp;
244  //pDst[2*(N-k)+1] = -(uIm + imTmp);
245  vtmp2q_f32 = vnegq_f32(vin1q2_f32.val[1]);
246  vtmp1q_f32 = vrev64q_f32(vin1q2_f32.val[0]);
247  vtmp2q_f32 = vrev64q_f32(vtmp2q_f32);
248

250  vin1q2_f32.val[1] = vcombine_f32(vget_high_f32(vtmp2q_f32),vget_low_f32(vtmp2q_f32));
251 
252  vtmpq2_f32 = vld2q_f32(pOut2+N);
253  vin1q2_f32.val[0] = vbslq_f32(vmask1q_u32,vtmpq2_f32.val[0],vin1q2_f32.val[0]);
254  vin1q2_f32.val[1] = vbslq_f32(vmask1q_u32,vtmpq2_f32.val[1],vin1q2_f32.val[1]);
255 
256  vst2q_f32(pOut2,vin2q2_f32);
257  vst2q_f32(pOut2+N,vin1q2_f32);
258  pOut2-=8;
259 
260 
261 
262  /*y[N/4] = a[N/4]-jb[N/4]; y[3*N/4] = a[N/4] + jb[N/4]*/

265  pDst[N+NBy2] = pSrc[NBy2];

327  pIn1+=8;

335  /*a[N/2-4] a[N/2-3] a[N/2-2] a[N/2-1]*/
336  vtmp1q_f32 = vcombine_f32(vget_high_f32(vtmp1q_f32),vget_low_f32(vtmp1q_f32));
337  /* b[N/2-4] b[N/2-3] b[N/2-2] b[N/2-1]*/
338  vtmp2q_f32 = vcombine_f32(vget_high_f32(vtmp2q_f32),vget_low_f32(vtmp2q_f32));
339

341  vureq_f32 = vaddq_f32(vin1q2_f32.val[0],vtmp1q_f32);
342  /*uIm = (b[k]-b[N/2-k])/2*/
343  vuimq_f32 = vsubq_f32(vin1q2_f32.val[1],vtmp2q_f32);
344 
345  /*VRe = (a[k]-a[N/2-k])/2*/
346  vvreq_f32 = vsubq_f32(vin1q2_f32.val[0],vtmp1q_f32);
347  /*Vim = (b[k]+b[N/2-k])/2*/
348  vvimq_f32 = vaddq_f32(vin1q2_f32.val[1],vtmp2q_f32);
349 
350  vureq_f32 = vmulq_f32(vureq_f32,vhalfq_f32);
351  vuimq_f32 = vmulq_f32(vuimq_f32,vhalfq_f32);
352  vvreq_f32 = vmulq_f32(vvreq_f32,vhalfq_f32);
353  vvimq_f32 = vmulq_f32(vvimq_f32,vhalfq_f32);
354 
355  vretwdq_f32 = vld1q_f32(pCoefRe);
356  vimtwdq_f32 = vld1q_f32(pCoefIm);
357  pCoefRe+=4;
358  pCoefIm+=4;
359 
360  /*reTmp = vRe*reTwd - vIm*imTwd */
361  vtmp1q_f32 = vmulq_f32(vvreq_f32,vretwdq_f32);
362  vtmp1q_f32 = vmlsq_f32(vtmp1q_f32,vvimq_f32,vimtwdq_f32);
363  /*imTmp = vIm*reTwd + vRe*imTwd */
364  vtmp2q_f32 = vmulq_f32(vvimq_f32,vretwdq_f32);
365  vtmp2q_f32 = vmlaq_f32(vtmp2q_f32,vvreq_f32,vimtwdq_f32);
366  //pDst[2*k] = uRe - imTmp;
367  //pDst[2*k+1] = uIm + reTmp;
368  vin1q2_f32.val[0] = vsubq_f32(vureq_f32,vtmp2q_f32);
369  vin1q2_f32.val[1] = vaddq_f32(vuimq_f32,vtmp1q_f32);
